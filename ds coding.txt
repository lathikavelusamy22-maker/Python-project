<script>

let tasks=[]

let quotes=[
"Success comes from hard work",
"Never give up",
"Dream big work hard",
"Small progress is still progress",
"Believe in yourself"
]

function login(){
if(user.value=="admin" && pass.value=="1234"){
loginPage.style.display="none"
todoPage.style.display="block"
load()
}else{
error.innerHTML="Invalid Login"
}
}

function logout(){
todoPage.style.display="none"
loginPage.style.display="block"
}

function addTask(){
if(task.value==""){
alert("Enter Task")
return
}

tasks.push({
text:task.value,
date:date.value || "No date",
time:time.value || "No time",
priority:priority.value,
progress:progress.value || 0
})

save()
show()

task.value=""
date.value=""
time.value=""
progress.value=""
}

function show(){
list.innerHTML=""

tasks.forEach((t,i)=>{

let li=document.createElement("li")

li.innerHTML=
"<b>"+t.text+"</b><br>"+
"Date: "+t.date+"<br>"+
"Time: "+t.time+"<br>"+
"Priority: "+t.priority+"<br>"+
"Progress: "+t.progress+"%"+

"<div class='taskButtons'>"+
"<button onclick='editTask("+i+")'>Edit</button>"+
"<button class='deleteBtn' onclick='deleteTask("+i+")'>Delete</button>"+
"</div>"

list.appendChild(li)

})
}

function editTask(i){
let t=tasks[i]

task.value=t.text
date.value=t.date
time.value=t.time
priority.value=t.priority
progress.value=t.progress

tasks.splice(i,1)
save()
show()
}

function deleteTask(i){
tasks.splice(i,1)
save()
show()
}

function save(){
localStorage.setItem("tasks",JSON.stringify(tasks))
}

function load(){
let data=localStorage.getItem("tasks")
if(data){
tasks=JSON.parse(data)
show()
}
}

function openDashboard(){
todoPage.style.display="none"
dashboardPage.style.display="block"

total.innerHTML=tasks.length
high.innerHTML=tasks.filter(t=>t.priority=="High").length
low.innerHTML=tasks.filter(t=>t.priority=="Low").length
}

function openQuotes(){
todoPage.style.display="none"
quotePage.style.display="block"
newQuote()
}

function newQuote(){
quoteText.innerHTML=quotes[Math.floor(Math.random()*quotes.length)]
}

function openProfile(){
todoPage.style.display="none"
profilePage.style.display="block"

profileName.innerHTML="admin"
profileTotal.innerHTML=tasks.length
profileCompleted.innerHTML=tasks.filter(t=>t.progress==100).length
profilePending.innerHTML=tasks.filter(t=>t.progress<100).length
}

function openSearch(){
todoPage.style.display="none"
searchPage.style.display="block"
searchTask()
}

function searchTask(){
let key=searchInput.value.toLowerCase()
searchList.innerHTML=""

tasks.filter(t=>t.text.toLowerCase().includes(key))
.forEach(t=>{
let li=document.createElement("li")
li.innerHTML=
"<b>"+t.text+"</b><br>"+
"Date: "+t.date+"<br>"+
"Time: "+t.time+"<br>"+
"Priority: "+t.priority+"<br>"+
"Progress: "+t.progress+"%"
searchList.appendChild(li)
})
}

function backTodo(){
dashboardPage.style.display="none"
quotePage.style.display="none"
profilePage.style.display="none"
searchPage.style.display="none"
todoPage.style.display="block"
}

</script>
