<script>

let users = [];
let otp = "";
let attempts = 0;
let time = 30;
let timerInterval;

// PAGE SWITCH
function showPage(page){
    document.querySelectorAll(".container").forEach(c=>c.classList.add("hidden"));
    document.getElementById(page).classList.remove("hidden");
}

// DARK MODE
function toggleDark(){
    document.body.classList.toggle("dark");
}

// REGISTER
function register(){
    let name = document.getElementById("name").value;
    let id = document.getElementById("id").value;

    if(users.find(u=>u.id==id)){
        regMsg.innerHTML="❌ Duplicate ID!";
        return;
    }

    users.push({name,id});
    regMsg.innerHTML="✅ Registered!";
}

// LOGIN
function login(){
    let id = document.getElementById("loginID").value;

    if(!users.find(u=>u.id==id)){
        loginMsg.innerHTML="❌ Not Found!";
        return;
    }

    otp = Math.floor(1000+Math.random()*9000);
    alert("OTP: "+otp);

    startTimer();
    showPage("otp");
}

// OTP TIMER
function startTimer(){
    time = 30;
    timerInterval = setInterval(()=>{
        document.getElementById("timer").innerText = "Time: "+time;

        time--;
        if(time<0){
            clearInterval(timerInterval);
            alert("OTP Expired!");
        }
    },1000);
}

// VERIFY OTP
function verifyOTP(){
    let val = document.getElementById("otpInput").value;

    if(val != otp){
        attempts++;
        otpMsg.innerHTML="❌ Wrong OTP";

        if(attempts>=3){
            alert("🚫 Fraud detected! Blocked.");
        }
        return;
    }

    clearInterval(timerInterval);
    showPage("face");
}

// FACE
function faceScan(){
    let ok = Math.random()>0.2;

    if(!ok){
        faceMsg.innerHTML="❌ Face mismatch";
        return;
    }

    faceMsg.innerHTML="✅ Face matched";
    showPage("finger");
}

// FINGERPRINT
function fingerScan(){
    setTimeout(()=>{
        fingerMsg.innerHTML="✅ Fingerprint matched";
        loadAdmin();
        showPage("admin");
    },1500);
}

// ADMIN LOAD
function loadAdmin(){
    let table = document.getElementById("data");
    table.innerHTML="";

    users.forEach((u,i)=>{
        table.innerHTML += `
        <tr>
        <td>${u.name}</td>
        <td>${u.id}</td>
        <td>
        <button onclick="del(${i})">Delete</button>
        <button onclick="edit(${i})">Edit</button>
        </td>
        </tr>`;
    });
}

// DELETE
function del(i){
    users.splice(i,1);
    loadAdmin();
}

// EDIT
function edit(i){
    let newName = prompt("New name:");
    if(newName){
        users[i].name = newName;
        loadAdmin();
    }
}

// SEARCH
function searchUser(){
    let val = document.getElementById("search").value.toLowerCase();

    let rows = document.querySelectorAll("#data tr");

    rows.forEach(r=>{
        r.style.display = r.innerText.toLowerCase().includes(val) ? "" : "none";
    });
}
